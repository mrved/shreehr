```prisma
model PollOption {

  order     Int            @default(0)

  @@map("poll_options")
}
```

**PollResponse model:**
```prisma
model PollResponse {
  id          String     @id @default(cuid())
  poll_id     String
  poll        Poll       @relation(fields: [poll_id], references: [id], onDelete: Cascade)
  option_id   String

  employee    Employee   @relation(fields: [employee_id], references: [id])
  created_at  DateTime   @default(now())

  @@map("poll_responses")
}
```

**Create `src/lib/email/templates/birthday-notification.ts`:**

This is a digest-style email sent to all employees listing today's birthdays and work anniversaries.

```typescript
interface BirthdayEntry {
  name: string;
  type: 'birthday' | 'anniversary';
  years?: number; // years of service for anniversaries
}

interface BirthdayNotificationData {
  recipientName: string;
  celebrations: BirthdayEntry[];
  dashboardUrl: string;
}

export function birthdayNotificationTemplate(data: BirthdayNotificationData) {
  const { recipientName, celebrations, dashboardUrl } = data;
  // Build a list of celebrations
  const celebrationLines = celebrations.map(c => {
    if (c.type === 'birthday') return `🎂 ${c.name} — Happy Birthday!`;
    return `🎉 ${c.name} — ${c.years} Year${c.years !== 1 ? 's' : ''} Work Anniversary!`;
  });

  return {
    subject: `[ShreeHR] Today's Celebrations`,
    html: `...styled HTML listing all celebrations...`,
    text: `Hi ${recipientName},\n\nToday we celebrate:\n${celebrationLines.join('\n')}\n\nView on dashboard: ${dashboardUrl}`,
  };
}
```
