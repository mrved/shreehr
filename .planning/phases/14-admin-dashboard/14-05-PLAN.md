**Check-in action:**
```typescript
const handleCheckIn = async () => {
  setLoading(true);
  try {
    const res = await fetch('/api/attendance/check-in', {
      method: 'POST',
      body: JSON.stringify({}), // remarks optional
      headers: { 'Content-Type': 'application/json' },
    });
    if (!res.ok) {
      const data = await res.json();
      setError(data.error || 'Failed to check in');
      return;
    }
    const attendance = await res.json();
    setAttendance({ check_in: attendance.check_in, check_out: null, status: attendance.status, work_minutes: null });
  } catch {
    setError('Network error');
  } finally {
    setLoading(false);
  }
};
```

**Check-out action:** Similar but POST to `/api/attendance/check-out`.
